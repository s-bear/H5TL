393  //if chunk_nbytes is 0, we should compute a value for it:
394  if(chunk_nbytes == 0) {
395  //get size of dataset in mb
396  size_t data_mb = (item_nbytes*util::product(data_shape.begin(),data_shape.end(),size_t(1))) >> 20;
397  data_mb = util::clip<size_t>(data_mb,1,1 << 23); //stay between 1 MB and 8 TB
398  //data size -> chunk size: 1 MB -> 1 line, 1 TB -> 1k lines
399  // 2^0 -> 2^0, 2^20 -> 2^10
400  // chunk_size = 2^(log2(data_mb)*10/20)*line_size = sqrt(data_mb)*line_size
401  chunk_nbytes = size_t(sqrt(data_mb))*line_nbytes;
402  }
403  //given chunk_nbytes, how many items should be in the chunk?
404  size_t desired_chunk_size(chunk_nbytes / item_nbytes);
405  //start with the chunk size the same as the whole shape:

407  //it's possible the shape is <= the desired chunk size
408  size_t chunk_size = util::product(chunk_shape.begin(),chunk_shape.end(),size_t(1));
409  if(chunk_size <= desired_chunk_size)

411  //if it's not, trim each dimension in turn to reduce the chunk size
